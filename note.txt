         create a tsconfig.json file
{
"compilerOptions": {

"outDir": "dist",
"strict": true,

"forceConsistentCasingInFileNames": true,

},
"include": ["src"]
}

"scripts": {
"dev": "nodemon --watch 'src/**/*.ts' --exec ts-node src/server.ts",
"build": "tsc",
"start": "node dist/server.js"
}
